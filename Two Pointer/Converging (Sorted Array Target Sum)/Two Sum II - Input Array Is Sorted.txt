var twoSum = function(numbers, target) {

    let l = 0
    let r = numbers.length-1

    while(l < r){
        let result = numbers[l] + numbers[r]

        if(result === target){
            return [l+1,r+1]
        }else if(result < target){
            l++
        }else{
            r--;
        }
    }  
};
